import java.util.Scanner;

// Custom Exception
class BookException extends Exception {

    BookException(String message) {

        super(message);
    }
}

// Book Class
class Book {

    int bookId;
    String title;
    boolean isIssued;

    // Constructor
    Book(int bookId, String title) {

        this.bookId = bookId;
        this.title = title;
        this.isIssued = false;
    }

    // Issue Book
    void issueBook() throws BookException {

        if (isIssued) {

            throw new BookException("Book Already Issued");

        } else {

            isIssued = true;

            System.out.println(title + " Book Issued Successfully");
        }
    }

    // Return Book
    void returnBook() throws BookException {

        if (!isIssued) {

            throw new BookException("Book is Not Issued");

        } else {

            isIssued = false;

            System.out.println(title + " Book Returned Successfully");
        }
    }

    // Display Book
    void displayBook() {

        System.out.println("\nBook ID: " + bookId);
        System.out.println("Book Title: " + title);

        if (isIssued) {

            System.out.println("Status: Issued");

        } else {

            System.out.println("Status: Available");
        }
    }
}

// Main Class
public class LibraryManagement {

    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);

        // Create Book
        Book b1 = new Book(101, "Java Programming");

        b1.displayBook();

        try {

            // Issue Book
            System.out.println("\nIssuing Book");

            b1.issueBook();

            // Issue Again
            System.out.println("\nIssuing Again");

            b1.issueBook();

        } catch (BookException e) {

            System.out.println("Exception: " + e.getMessage());
        }

        try {

            // Return Book
            System.out.println("\nReturning Book");

            b1.returnBook();

        } catch (BookException e) {

            System.out.println("Exception: " + e.getMessage());
        }

        b1.displayBook();

        sc.close();
    }
}

AsyncService.java

package com.example.asyncdemo;

import org.springframework.scheduling.annotation.Async;
import org.springframework.stereotype.Service;

@Service
public class AsyncService {

    @Async
    public void processTask() {

        System.out.println("Task Started: " + Thread.currentThread().getName());

        try {
            Thread.sleep(5000);
        } catch (InterruptedException e) {
            e.printStackTrace();
        }

        System.out.println("Task Completed: " + Thread.currentThread().getName());
    }
}

AsyncController.java

package com.example.asyncdemo;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

@RestController
public class AsyncController {

    @Autowired
    private AsyncService asyncService;

    @GetMapping("/start")
    public String startTask() {

        asyncService.processTask();

        return "Async Task Started...";
    }
}
